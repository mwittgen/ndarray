100 int level
101 ) {
102 os << options._open;

106 Formatter<typename ExpressionBase<Derived>::Reference>::apply(options,os,*iter,level+1);
107 for (++iter; iter != end; ++iter) {
108 os << options._delimiter;
109 os << std::endl << std::string(level,' ');
110 Formatter<typename ExpressionBase<Derived>::Reference>::apply(options,os,*iter,level+1);
111 }

113 os << options._close;
